/* ====================================================================
   TSL demo fixtures — REAL ProofBundleV1 files generated by the repo's
   CLI (npm run demo / revoke-key). They are verified by the SAME
   in-browser verifyTSL bundle — demo mode still does real verification.
   ==================================================================== */
(function (T) {
  'use strict';
  // chip key → { label, file, note }
  T.FIXTURES = [
    { key: 'verified', label: 'A verified proof', file: 'fixtures/_demo.bundle.json',
      note: 'A real signed event, recorded in the log and anchored to a checkpoint.' },
    { key: 'revoked', label: 'A revoked key', file: 'fixtures/_revoked.bundle.json',
      note: 'The same identity after its signing key was cancelled — verification genuinely fails.' }
  ];
  T._fixtureCache = {};
  T.loadFixture = function (key) {
    var f = T.FIXTURES.filter(function (x) { return x.key === key; })[0];
    if (!f) return Promise.reject(new Error('unknown fixture'));
    if (T._fixtureCache[key]) return Promise.resolve(T._fixtureCache[key]);
    return fetch(f.file).then(function (r) { if (!r.ok) throw new Error('fixture ' + r.status); return r.json(); })
      .then(function (bundle) { T._fixtureCache[key] = bundle; return bundle; });
  };
})(window.TSL);
